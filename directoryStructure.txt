LandAdjacency_Serial/
├── data/
│   ├── raw/                   # 保留原始土地利用矢量数据（gzlanduse_2013~2023.shp）备份（可选）
│   └── landuse.gpkg           # ⭐ 集中存储所有年份图层
├── output/                    # 存放输出图层（融合矢量、栅格图）
├── config/                    # 存放配置项，如地类编码映射、权重等
│   ├── class_map.json         # 地类编码表，如 {1: "农田", 2: "森林", ...}
│   ├── weight_config.json     # 指标权重配置
│   ├── config.yaml
├── utils/                     # 工具函数，如投影转换、日志记录
│   ├── geodata_io.py          # ⭐ 重命名！统一读写接口（支持GPKG）
│   ├── spatial_index.py       # ⭐ 新增！空间索引创建函数
│   ├── config_reader.py
│   ├── timer.py
├── core/                      # 核心算法模块，每步一个.py文件
│   ├── preprocess.py          # S1：投影、标准化
│   ├── classify_landuse.py    # S2.1：分类
│   ├── extract_features.py    # S2.2：边界与重心提取
│   ├── build_laim.py          # S3：构建 LAIM 图谱
│   ├── build_lcsm.py          # S4：构建 LCSM 图谱
│   ├── visualize_matrix.py    # 独立脚本：绘制 LAIM 和 LCSM 热力图
│   ├── adjacency_detector.py  # S5：识别邻接与非邻接(从这一步开始输入文件只有目标期的2024年)
│   ├── compute_effect.py      # S6–S8：缓冲、相交、影响赋值
│   ├── merge_and_raster.py    # S9–S10：图层合并 + 栅格化
├── main.py                    # 主程序入口，串联全流程
└── requirements.txt           # 所需依赖库
